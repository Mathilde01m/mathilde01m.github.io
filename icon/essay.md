      document.addEventListener("DOMContentLoaded", function () {
        // Définition des couleurs pour DPE
        const colorMappingDPE = {
          A: "#009933", // Vert foncé
          B: "#33cc33", // Vert clair
          C: "#ffff00", // Jaune
          D: "#ffcc00", // Orange clair
          E: "#ff6600", // Orange foncé
          F: "#ff3300", // Rouge clair
          G: "#cc0000", // Rouge foncé
        };

        // Définition des couleurs pour GES (ton violets)
        const colorMappingGES = {
          A: "#FCF0F4", // Violet foncé
          B: "#D9C1DB", // Violet clair
          C: "#9E75AC", // Violet moyen
          D: "#835A9C", // Violet rose
          E: "#663B8D", // Rose foncé
          F: "#835A9C", // Rose vif
          G: "#6A418F", // Rose très foncé
        };

        // Fonction pour appliquer les couleurs et le format aux boxes DPE et GES
        function updateBoxes(selector, label, colorMapping) {
          document.querySelectorAll(selector).forEach((box) => {
            const letter = box.textContent.trim().toUpperCase();

            if (colorMapping[letter]) {
              // Appliquer la couleur de fond selon la lettre
              box.style.backgroundColor = colorMapping[letter];

              // S'assurer que le texte est lisible (noir pour couleurs claires, blanc pour couleurs foncées)
              box.style.color =
                letter === "A" || letter === "B" || letter === "C"
                  ? "#000"
                  : "#fff";

              // Ajouter "DPE" ou "GES" devant la lettre
              box.textContent = `${label} ${letter}`;
            }
          });
        }

        // Appliquer la mise en forme pour les DPE et GES
        updateBoxes(".dpe-box", "DPE", colorMappingDPE);
        updateBoxes(".ges-box", "GES", colorMappingGES);
      });

      function updateBorders() {

        // Pour chaque boîte, changer la bordure en fonction de la valeur de "data-value"

          const value = parseInt(box.getAttribute("data-value")); // Récupérer la valeur de data-value
          box.classList.remove(

            "border-1",
            "border-2",
            "border-3",
            "border-4",
            "border-5"
          ); // Retirer les classes précédentes
          box.classList.add("border-" + value); // Appliquer la nouvelle bordure
        });
        center.forEach((box) => {
          const value = parseInt(box.getAttribute("data-value")); // Récupérer la valeur de data-value

            "border-4",

          ); // Retirer les classes précédentes

        });
